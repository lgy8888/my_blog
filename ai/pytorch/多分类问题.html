def train(train_dl,model,loss_fn,opt):
    size = len(train_dl.dataset) # è·å–å½“å‰æ•°æ®é›†çš„å¤§å°
    num_batches = len(train_dl) # è·å–æ€»æ‰¹æ¬¡æ•°
    train_loss, correct = 0, 0 # åˆå§‹åŒ–å‡†ç¡®ç‡å’Œå¹³å‡è®­ç»ƒæŸå¤±
    model.train() # å°†æ¨¡å‹è®¾ç½®ä¸ºè®­ç»ƒæ¨¡å¼

    for x, y in train_dl: 
        x, y = x.to(device), y.to(device) # å°†æ•°æ®é›†æ·»åŠ åˆ°è®¾å¤‡
        pred = model(x)
        loss = loss_fn(pred, y) # loss_fnè¿”å›çš„æ˜¯è¯¥æ‰¹æ¬¡ä¸­æ‰€æœ‰æ ·æœ¬çš„å¹³å‡æŸå¤±
        opt.zero_grad()
        loss.backward()

            # å°†æ¯ä¸ªbatché‡Œé¢é¢„æµ‹æ­£ç¡®çš„æ ·æœ¬æ•°ç›®ä¹‹å’Œç´¯åŠ åˆ°correctä¸­

            # å°†æ¯ä¸ªbatchçš„æŸå¤±ç´¯åŠ åˆ°train_lossä¸­
            train_loss += loss.item()

    correct /= size # é¢„æµ‹æ­£ç¡®/å…¨éƒ¨ -> å‡†ç¡®ç‡
    train_loss /= num_batches # å¹³å‡æŸå¤±ä¹‹å’Œ/æ‰¹æ¬¡æ•° -> å…¨å±€å¹³å‡æŸå¤±
    return correct, train_loss

def test(test_dl,model,loss_fn):
    size = len(test_dl.dataset) # è·å–å½“å‰æ•°æ®é›†çš„å¤§å°
    num_batches = len(test_dl) # è·å–æ€»æ‰¹æ¬¡æ•°
    test_loss, correct = 0, 0 # åˆå§‹åŒ–å‡†ç¡®ç‡å’Œå¹³å‡è®­ç»ƒæŸå¤±
    model.eval() # å°†æ¨¡å‹è®¾ç½®ä¸ºæµ‹è¯•æ¨¡å¼

        for x, y in test_dl:

            pred = model(x)

            # å°†æ¯ä¸ªbatché‡Œé¢é¢„æµ‹æ­£ç¡®çš„æ ·æœ¬æ•°ç›®ä¹‹å’Œç´¯åŠ åˆ°correctä¸­

            test_loss += loss.item()

def fit(epochs, train_dl, test_dl, model, loss_fn, opt):
    train_loss = []
    train_acc = []
    test_loss = []
    test_acc = []
    
    for epoch in range(epochs):
        epoch_train_acc, epoch_train_loss = train(train_dl, model, loss_fn, opt)
        epoch_test_acc, epoch_test_loss = test(test_dl, model, loss_fn)
        
        train_acc.append(epoch_train_acc)

        template = ("epoch:{:2d}, train_loss:{:.5f}, train_acc:{:.1f}%, test_loss:{:.5f}, test_acc:{:.1f}%")
        print(template.format(epoch, epoch_train_loss, epoch_train_acc*100, 
                              epoch_test_loss,epoch_test_acc*100))
        
    print('Done')
    return train_loss, train_acc, test_loss, test_acc
