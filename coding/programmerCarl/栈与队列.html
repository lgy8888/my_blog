                                              # 栈与队列

                                              # 基本原理

                                              • 栈是先进后出

                                              • 队列是先进先出

                                              • 栈提供push和pop等接口，所有元素必须符合先进后出规则

                                              • 栈不提供走访功能，也不提供迭代器

                                              • 栈是以底层容器来完成所有工作的，对外提供统一接口，底层容器是可插拔的（意思是可以控制是哦哪种容器来实现栈的功能）

                                              # 单栈

                                              # 基本思路

                                              使用LinkedList双端链表来模拟栈，pop -> removeFirst，push -> addFirst，peek -> getFirst

                                              修改push方法的逻辑，使其变成先进先出

                                              # 代码实现

                                              class MyQueue {
    LinkedList<Integer> stack = new LinkedList<>();

    public MyQueue() {

    }

    public void push(int x) {
        List<Integer> tmp = new ArrayList<>();
        while (!stack.isEmpty()) {
            tmp.add(stack.removeFirst());
        }
        stack.addFirst(x);
        for (int i = tmp.size() - 1; i >= 0; i--) {
            stack.addFirst(tmp.get(i));
        }
    }

    public int pop() {
        return stack.removeFirst();
    }

    public int peek() {
        return stack.getFirst();
    }

    public boolean empty() {
        return stack.isEmpty();
    }
}

                                              # 双栈

                                              # 基本思路

                                              使用LinkedList双端链表来模拟栈，pop -> removeFirst，push -> addFirst，peek -> getFirst

                                              定义两个栈，一个inStack模拟入队，一个outStack模拟出队

                                              push入队：inStack执行push

                                              pop出队：判断outStack是否有元素

                                              • 如果有，直接pop
                                              • 如果没有，将inStack的元素push过来

                                              peek获取队头元素：判断outStack是否有元素

                                              • 如果有，直接peek
                                              • 如果没有，将inStack的元素push过来

                                              empty队列是否为空：判断栈是否为空

                                              # 代码实现

                                              class MyQueue {
    LinkedList<Integer> inStack = new LinkedList<>();
    LinkedList<Integer> outStack = new LinkedList<>();

    public MyQueue() {

    }

    public void push(int x) {
        inStack.addFirst(x);
    }

    public int pop() {
        if (outStack.isEmpty()){
            in2Out();
        }
        return outStack.removeFirst();
    }

    public int peek() {
        if (outStack.isEmpty()){
            in2Out();
        }
        return outStack.getFirst();
    }

    public boolean empty() {
        return inStack.isEmpty() && outStack.isEmpty();
    }

    private void in2Out(){
        while(!inStack.isEmpty()){
            outStack.push(inStack.pop());
        }
    }
}

                                              # 用队列实现栈

                                              # 基本思路

                                              使用LinkedList双端链表来模拟队列，pop -> removeFirst，push -> addLast，top-> getFirst

                                              修改push方法的逻辑，使其变成先进后出

                                              # 代码实现

                                              class MyStack {
    LinkedList<Integer> queue = new LinkedList<>();

    public MyStack() {

    }

    public void push(int x) {
        List<Integer> tmp = new ArrayList<>();
        while (!queue.isEmpty()) {
            tmp.add(queue.removeFirst());
        }
        queue.addLast(x);
        for (Integer t : tmp) {
            queue.addLast(t);
        }
    }

    public int pop() {
        return queue.removeFirst();
    }

    public int top() {
        return queue.getFirst();
    }

    public boolean empty() {
        return queue.isEmpty();
    }
}

                                              # 有效的括号

                                              力扣题目链接open in new window

                                              # 基本思路

                                              定义一个栈来接收左括号

                                              定义一个左括号右括号的映射表

                                              遍历字符串，若遍历到左括号则入栈，若遍历到右括号则弹栈，弹出来的元素与遍历到的元素比较是否匹配

                                              # 代码实现

                                              class Solution {
    public boolean isValid(String s) {
        LinkedList<Character> stack = new LinkedList<>();
        Map<Character, Character> map = new HashMap<>();
        boolean res = false;
        map.put('(', ')');
        map.put('[', ']');
        map.put('{', '}');
        if (!map.containsKey(s.charAt(0))) return false;
        for (char c : s.toCharArray()) {
            if (map.containsKey(c)) {
                stack.addFirst(c);
            } else {
                if (stack.isEmpty()) return false;
                else if (c != map.get(stack.removeFirst())) return false;
            }
        }
        return stack.isEmpty();
    }
}

                                              # 删除字符串中的所有相邻重复项

                                              力扣题目链接open in new window

                                              # 基本思想

                                              跟有效括号差不多

                                              定义一个栈

                                              字符串从尾遍历入栈

                                              每次入栈时判断栈顶元素是否相同

                                              相同就弹出，不相同就加入

                                              整个字符串都入栈完毕之后

                                              弹出来做结果返回

                                              # 代码实现

                                              class Solution {
    public String removeDuplicates(String s) {
        LinkedList<Character> stack = new LinkedList<>();
        char[] carr = s.toCharArray();
        for(int i = carr.length-1; i >= 0; i--){
            char c = carr[i];
            if(stack.isEmpty()){
                stack.addFirst(c);
            }else{
                if(stack.getFirst() == c){
                    stack.removeFirst();
                }else{
                    stack.addFirst(c);
                }
            }
        }
        StringBuilder res = new StringBuilder();
        while(!stack.isEmpty()){
            res.append(stack.removeFirst());
        }
        return res.toString();
    }
}

                                              # 逆波兰表达式求值

                                              力扣题目链接open in new window

                                              逆波兰表达式：是一种后缀表达式，所谓后缀就是指运算符写在后面。

                                              中缀转后缀——( 1 + 2 ) * ( 3 + 4 )

                                              1. 给中缀表达式的每个子表达式套上括号——( ( 1 + 2 ) * ( 3 + 4 ) )
                                              2. 给子表达式的运算顺序排序——括号优先，所以两个+并列第一，*第二
                                              3. 后运算的运算符先移动
                                                • ( ( 1 + 2 ) ( 3 + 4 ) * )
                                                • ( ( 1 2 + ) ( 3 4 + ) * )

                                              逆波兰表达式主要有以下两个优点：

                                              • 去掉括号后表达式无歧义，上式即便写成 1 2 + 3 4 + * 也可以依据次序计算出正确结果。
                                              • 适合用栈操作运算：遇到数字则入栈；遇到运算符则取出栈顶两个数字进行计算，并将结果压入栈中。

                                              # 基本思路

                                              定义一个栈

                                              遍历给定的字符串数组

                                              遇到数字就入栈

                                              遇到运算符就连弹出两个出来运算——ab+ -> a+b

                                              将算好的值入栈

                                              返回栈剩下的最后一个元素，就是该后缀表达式的值

                                              # 易错点

                                              • 减法运算和除法运算由于被减数和减数与被除数和除数的差异，需要特殊处理
                                                • 栈弹出来的元素是倒序的，需要连弹的两个元素做个倒换才能做减法和除法

                                              # 代码实现

                                              class Solution {
    public int evalRPN(String[] tokens) {
        Deque<Integer> stack = new LinkedList<>();
        for(String s : tokens){
            switch (s) {
                case "+":
                    stack.push(stack.pop() + stack.pop());
                    break;
                case "-": {
                    int b = stack.pop();
                    int a = stack.pop();
                    stack.push(a - b);
                    break;
                }
                case "*":
                    stack.push(stack.pop() * stack.pop());
                    break;
                case "/": {
                    int b = stack.pop();
                    int a = stack.pop();
                    stack.push(a / b);
                    break;
                }
                default:
                    stack.push(Integer.valueOf(s));
                    break;
            }
        }
        return stack.pop();
    }
}

                                              # 滑动窗口最大值

                                              力扣题目链接open in new window

                                              # 暴力解法

                                              # 基本思路

                                              遍历nums数组

                                              遍历的过程中维护一个最大值max

                                              每遍历k次就将最大值放进res中

                                              然后将指针i回退到滑动窗口头，并且计数和max重置

                                              # 代码实现

                                              class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
        if (nums.length <= 1) return nums;
        int i = 0;
        int max = nums[0];
        List<Integer> res = new ArrayList<>();
        int count = 0;
        while (i < nums.length) {
            max = Math.max(nums[i], max);
            i++;
            count++;
            if (count == k) {
                res.add(max);
                i = i - k + 1;
                count = 0;
                max = Integer.MIN_VALUE;
            }
        }
        int[] resArr = new int[res.size()];
        for (int a = 0; a < res.size(); a++) {
            resArr[a] = res.get(a);
        }
        return resArr;
    }
}

                                              # 单调队列

                                              # 基本思路

                                              如何让指针不回退——只遍历一次，线性时间复杂度

                                              自定义一个单调队列，实现滑动窗口

                                              • 实现pop(int x)，push(int x)，top()三个方法，分别对应弹出指定元素，加入指定元素，获取队头元素
                                              • 维护单调队列的头部元素始终是队中最大的元素

                                              每次入队时，从队尾开始向队头出发，清理沿途所有比传入值小的元素，清理完将指定元素入队

                                              每次出队时，判断队头元素是否是指定元素

                                              • 如果是，则说明该最大元素已经到滑动窗口的边界，需要出队
                                              • 如果不是，则说明该元素不是队中最大元素，已经被清理掉了

                                              主方法中，定义单调队列

                                              遍历nums数组，将数组元素逐一入队

                                              前k-1个元素(滑动窗口初始化)不需要执行出队

                                              后面的元素每入队一个就出队一个，维持窗口长度为k个元素

                                              滑动窗口初始化之后每遍历一次就将队头元素获取出来加入结果数组中

                                              # 代码实现

                                              class Solution {
    class MyQueue {
        private final Deque<Integer> queue;

        public MyQueue() {
            queue = new LinkedList<>();
        }

        // 传入要入队元素，从队尾开始，清除所有比该元素小的
        public void push(int val) {
            while ((!queue.isEmpty()) && val > queue.getLast()) {
                queue.removeLast();
            }
            queue.addLast(val);
        }

        // 传入要弹出的元素，判断如果是队头元素，则弹出，不是则说明该元素已经被弹出了
        public void pop(int val) {
            if (!queue.isEmpty() && val == queue.getFirst()) {
                queue.removeFirst();
            }
        }

        // 获取队头元素
        public int top() {
            return queue.getFirst();
        }
    }

    public int[] maxSlidingWindow(int[] nums, int k) {
        MyQueue queue = new MyQueue();
        int index = 0;
        int[] res = new int[nums.length - k + 1];
        for (int i = 0; i < nums.length; i++) {
            if (i >= k - 1) {
                queue.push(nums[i]);
                res[index++] = queue.top();
                queue.pop(nums[i - k + 1]);
            } else {
                
                queue.push(nums[i]);
            }
        }
        return res;
    }
    
}

                                              # 前K个高频元素

                                              力扣题目链接open in new window

                                              # 基本思想

                                              1. 对给定数组进行计数并存入map
                                              2. 对map进行按照value进行排序
                                                • 排序可以用暴力排序
                                                • 最优解使用优先级队列进行排序——大顶堆 / 小顶堆
                                              3. 取出排完序的前k个key

                                              # 易错点

                                              • 大根堆是最大的在根节点
                                                • 使用大根堆就将所有元素都加进堆中，最后返回前K个节点
                                              • 小根堆是最小的在根节点
                                                • 使用小根堆就只需要维护前k个元素在堆里，最后将堆中元素全量返回

                                              # 代码实现

                                              /*Comparator接口说明:
 * 返回负数，形参中第一个参数排在前面；返回正数，形参中第二个参数排在前面
 * 对于队列：排在前面意味着往队头靠
 * 对于堆（使用PriorityQueue实现）：从队头到队尾按从小到大排就是最小堆（小顶堆），
 *                                从队头到队尾按从大到小排就是最大堆（大顶堆）--->队头元素相当于堆的根节点
 * */
class Solution {
    //解法1：基于大顶堆实现
    public int[] topKFrequent1(int[] nums, int k) {
        Map<Integer,Integer> map = new HashMap<>();//key为数组元素值,val为对应出现次数
        for(int num:nums){
            map.put(num,map.getOrDefault(num,0)+1);
        }
        //在优先队列中存储二元组(num,cnt),cnt表示元素值num在数组中的出现次数
        //出现次数按从队头到队尾的顺序是从大到小排,出现次数最多的在队头(相当于大顶堆)
        PriorityQueue<int[]> pq = new PriorityQueue<>((pair1, pair2)->pair2[1]-pair1[1]);
        for(Map.Entry<Integer,Integer> entry:map.entrySet()){//大顶堆需要对所有元素进行排序
            pq.add(new int[]{entry.getKey(),entry.getValue()});
        }
        int[] ans = new int[k];
        for(int i=0;i<k;i++){//依次从队头弹出k个,就是出现频率前k高的元素
            ans[i] = pq.poll()[0];
        }
        return ans;
    }
    //解法2：基于小顶堆实现
    public int[] topKFrequent2(int[] nums, int k) {
        Map<Integer,Integer> map = new HashMap<>();//key为数组元素值,val为对应出现次数
        for(int num:nums){
            map.put(num,map.getOrDefault(num,0)+1);
        }
        //在优先队列中存储二元组(num,cnt),cnt表示元素值num在数组中的出现次数
        //出现次数按从队头到队尾的顺序是从小到大排,出现次数最低的在队头(相当于小顶堆)
        PriorityQueue<int[]> pq = new PriorityQueue<>((pair1,pair2)->pair1[1]-pair2[1]);
        for(Map.Entry<Integer,Integer> entry:map.entrySet()){//小顶堆只需要维持k个元素有序
            if(pq.size()<k){//小顶堆元素个数小于k个时直接加
                pq.add(new int[]{entry.getKey(),entry.getValue()});
            }else{
                if(entry.getValue()>pq.peek()[1]){//当前元素出现次数大于小顶堆的根结点(这k个元素中出现次数最少的那个)
                    pq.poll();//弹出队头(小顶堆的根结点),即把堆里出现次数最少的那个删除,留下的就是出现次数多的了
                    pq.add(new int[]{entry.getKey(),entry.getValue()});
                }
            }
        }
        int[] ans = new int[k];
        for(int i=k-1;i>=0;i--){//依次弹出小顶堆,先弹出的是堆的根,出现次数少,后面弹出的出现次数多
            ans[i] = pq.poll()[0];
        }
        return ans;
    }
}
