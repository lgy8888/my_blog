                                      class Solution {
    public int[][] generateMatrix(int n) {
        int i = 0;
        int j = 0;
        int[][] res = new int[n][n];
        int start = 0;
        int val = 1;
        while (val < n * n) {
            int end = n - start - 1;
            for (i = start; i < end; i++) {
                res[start][i] = val++;
            }
            for (j = start; j < end; j++) {
                res[j][end] = val++;
            }
            for (i = end; i > start; i--) {
                res[end][i] = val++;
            }
            for (j = end; j > start; j--) {
                res[j][start] = val++;
            }
            start++;
        }
        if (n % 2 == 1) {
            res[n/2][n/2] = n*n;
        }
        return res;
    }
}
