                                      # 数组

                                      # 基本原理

                                      • 数组是存放在连续内存空间上的相同类型数据的集合

                                      • 数组可以方便的通过下标索引的方式获取到下标下对应的数据

                                      • 数组下标是从0开始的

                                      • 数组内存空间的地址是连续的

                                        • 所以我们删除或增加元素的时候需要移动其他元素的地址

                                      • 数组元素是不能删除的，只能覆盖

                                      # 二分查找

                                      力扣题目链接open in new window

                                      # 基本逻辑

                                      • 定义左边界left，右边界right
                                      • while循环遍历搜索区间
                                        • 定义中间指针mid
                                        • 判断中间指针指向的值与target的关系
                                          • 中间值大于target则更新右边界
                                          • 中间值小于target则更新左边界
                                          • 中间值等于target则返回中间指针mid
                                      • while循环结束，找不到target，返回-1

                                      # 易错点

                                      • while判断条件：left < right or left <= right
                                      • 边界处理条件：right = middle or middle - 1

                                      判断while循环条件的方法：left和right的关系是否符合区间定义

                                      **判断边界处理条件的方法：**根据区间右侧是否为闭

                                      # 左闭右闭

                                      • [left, right]

                                      • while循环条件：left <= right合法，选择它

                                      • 边界处理条件：右闭，右边界包含在区间里面，不能对其进行二次搜索。

                                        • 所以对右边界进行更新的时候不能将middle包含在内
                                        • 所以选择right = middle - 1
                                      int left = 0;
int right = nums.length - 1;
while(left <= right){
  int mid = (left + right) / 2;
  if(nums[mid] < target){
      left = mid + 1;
  }else if(target < nums[mid]){
      right = mid - 1;
  }else{
    return mid;
  }
}
return -1;

                                      # 左闭右开

                                      • [left, right)
                                      • while循环条件：left <= right不合法， 选择left < right
                                      • 边界选择条件：右开，右边界不包含在区间里面，需要对其进行搜索。
                                        • 所以对右边界进行更新的时候需要将middle包含在内
                                        • 所以选择right = middle

int right = nums.length;
while(left < right){
  int mid = (left + right) / 2;
  if(nums[mid] < target){
      left = mid + 1;
  }else if(target < nums[mid]){
      right = mid - ;
  }else{
    return mid;
  }
}
return -1;

                                      # 移除元素

                                      力扣题目链接open in new window

                                      # 暴力解法

                                      # 基本思路

                                      两层for，一层遍历数组，一层更新数组

                                      # 易错点

                                      • 内层for循环中关于边界的设定(因为有i+1索引)
                                        • 若定义j = i，则j < len - 1
                                        • 若定义j = i + 1，则j < len
                                      • 内层for循环结束后需要跟新外层指针i和数组长度len

                                      # 代码实现

                                      class Solution {
    public int removeElement(int[] nums, int val) {
        int len = nums.length; // 定义最后要返回的数组长度值
        for(int i = 0; i < len; i++){
            // 找到要删除的值
            if(val == nums[i]){
                // 遍历该值后面的所有值往前移动，覆盖前值
                for(int j = i; j < len-1; j++){
                    nums[j] = nums[j+1];
                }
                // 删除完成后更新外层指针i和数组当前长度len
                i--;
                len--;
            }
        }
        return len;
    }
}
                                      • 时间复杂度：$O(n^2)$
                                      • 空间复杂度：$O(1)$

                                      # 双指针

                                      # 基本思路

                                      定义快慢两个指针

                                      快指针用来获取新数组中的元素

                                      慢指针用来获取新数组中需要更新的位置

                                      新数组指不包含目标值的数组

                                      最后返回新数组长度，也就是慢指针的值

                                      # 易错点

                                      • 慢指针用变量定义，快指针用循环定义
                                        • 快指针是无论如何都要往前走的
                                        • 慢指针走不走取决于是否需要更新新数组的值，也就是快指针指向的元素是不是待删除的目标值
                                      • 最后返回慢指针就好，不需要+1

                                      # 代码实现

                                      class Solution {
    public int removeElement(int[] nums, int val) {
        // 定义慢指针
        int slow = 0; 
        // 定义快指针，使用快指针遍历目标数组
        for(int fast = 0; fast < nums.length; fast++){
            // 遇到不等于目标值的数组元素就赋值给新数组
            if(val != nums[fast]){
                // slow++：取索引为slow的值，取完slow+1
                nums[slow++] = nums[fast];
            }
        }
        return slow;
    }
}
                                      • 时间复杂度：$O(n)$
                                      • 空间复杂度：$O(1)$

                                      # 有序数组的平方

                                      力扣题目链接open in new window

                                      # 暴力解法

                                      # 实现思路

                                      遍历数组将每个数平方，用快排排序

                                      性能取决于快排

                                      # 代码实现

                                      class Solution {
    public int[] sortedSquares(int[] nums) {
        for(int i = 0; i < nums.length; i++){
            nums[i] = nums[i]*nums[i];
        }
        Arrays.sort(nums);
        return nums;
    }
}

                                      # 双指针法

                                      # 基本思路

                                      给定数组是有序的，平方后原本负数的值可能会变成大值

                                      所以数组平方后的最大值就只能出现在最左或最右

                                      定义两个指针，一个左指针i，一个右指针j

                                      再定义一个新数组res，跟给定数组一样大，定义一个从后往前迭代的指针k

                                      比较左值平方和右值平方，给res[k]赋较大的值，然后k--

                                      # 代码实现

                                      class Solution {
    public int[] sortedSquares(int[] nums) {
        // 定义返回数组
        int[] res = new int[nums.length];
        // 定义指针k，用于给res数组赋值
        int k = res.length - 1;
        // 定义右指针j
        int j = nums.length - 1;
        // 定义左指针i
        int i = 0;
        // 遍历给定数组
        while(i <= j){
            int i2 = nums[i]*nums[i];
            int j2 = nums[j]*nums[j];
            // 给res数组赋平方的较大值并迭代
            if(i2 < j2){
                res[k--] = j2;
                j--;
            }else{
                res[k--] = i2;
                i++;
            }
        }
        return res;
    }
}
                                      • 时间复杂度：$O(n)$

                                      # 长度最小的子数组

                                      力扣题目链接open in new window

                                      # 暴力解法

                                      # 实现逻辑

                                      • 定义result变量，取int类型的最大数
                                      • 定义两层循环，i控制遍历整个数组，j控制每次遍历取子序列之和
                                      • 内层循环中设置条件，若子序列之和大于等于target值则将子序列长度与result值作比较，将小值覆盖到result上然后break
                                      • 外层循环结束后便可以得出长度最小的子数组长度
                                      • 若result未改变，则说明数组元素总和达不到target值，则返回0，否则返回result

                                      # 代码实现

                                      class Solution {
    public int minSubArrayLen(int target, int[] nums) {
        int result = Integer.MAX_VALUE;
        int subLength = 0;
        for(int i = 0; i < nums.length; i++){
            int sum = 0;
            for(int j = i; j < nums.length; j++){
                sum += nums[j];
                if(sum >= target){
                    subLength = j+1-i;
                    result = result < subLength ? result : subLength;
                    break;
                }
            }
        }
        return result == Integer.MAX_VALUE ? 0 : result;
    }
}

                                      # 滑动窗口

                                      # 基本思想

                                      滑动窗口跟双指针类似，通过控制前后两指针来形成中间的子序列（滑动窗口）

                                      所谓滑动窗口，就是不断调节子序列的起始位置和终止位置，从而得出想要的结果

                                      使用滑动窗口，主要确定三点：

                                      • 窗口内是什么
                                        • 满足子序列之和大于等于target值的最小连续数组
                                      • 如何移动窗口的起始位置
                                        • 窗口之和的值>=target了，窗口就该移动了（使其子序列不符合条件）
                                      • 如何移动窗口的结束位置
                                        • 窗口的结束位置就是遍历数组的指针

                                      # 实现逻辑

                                      • 定义前指针i和后指针j
                                      • 后指针j先移动，寻找符合条件的子序列
                                      • 找到后移动前指针i，移动到子序列不符合条件为止，继续移动后指针j
                                      • 重复循环至循环结束

                                      # 易错点

                                      • result变量需要定义一个最大值
                                      • 判断与暴力解法不同使用了while而不是if是因为
                                        • 前指针i的移动不是一蹴而就的，可能移动几次都没办法使得子序列不符合条件，需要多次移动，所以用while

                                      # 代码实现

                                      class Solution {
    public int minSubArrayLen(int target, int[] nums) {
        // 定义返回值，int类型的最大值
        int result = Integer.MAX_VALUE;
        // 定义子序列之和
        int sum = 0;
        // 定义子序列的长度
        int subLength = 0;
        // 定义前指针
        int i = 0;
        // 定义后指针，循环遍历给定数组
        for(int j = 0; j < nums.length; j++){
            // 累加子序列内的值
            sum += nums[j];
            // 若子序列内值之和大于target（符合条件）
            while(sum >= target){
                // 开始移动前指针
                subLength = j - i + 1; 					// 获取子序列长度
                result = Math.min(subLength, result); 	// 更新返回值，取原返回值与当前子序列长度中的小值
                sum -= nums[i];							// 缩小滑动窗口
                i++;									// 更新前指针
            }
        }
        return result == Integer.MAX_VALUE ? 0 : result;
    }
}

                                      # 螺旋矩阵

                                      力扣题目链接open in new window

                                      # 基本思路

                                      模拟旋转的过程

                                      定义两个指针，行指针i和列指针j

                                      在纸上模拟最小两层结构(4和5)的旋转，将循环过程中的具体变量值写出来

                                      找循环过程中规律一致的变量

                                      找出来有三个：start，end，val

                                      start：表示循环的圈数，从0开始

                                      end：表示每次循环赋值的边长

                                      val：每次循环赋的值

                                      定义外层循环 ，条件为val< $n^2$

                                      定义内层循环，遍历给四条边赋值

                                      • 根据前面纸上模拟的边界条件来定义内层循环

                                      最后看n取单数还是双数来判断是否需要给最后一个值赋值

                                      # 代码实现

                                      class Solution {
    public int[][] generateMatrix(int n) {
        int i = 0;
        int j = 0;
        int[][] res = new int[n][n];
        int start = 0;
        int val = 1;
        while (val < n * n) {
            int end = n - start - 1;
            for (i = start; i < end; i++) {
                res[start][i] = val++;
            }
            for (j = start; j < end; j++) {
                res[j][end] = val++;
            }
            for (i = end; i > start; i--) {
                res[end][i] = val++;
            }
            for (j = end; j > start; j--) {
                res[j][start] = val++;
            }
            start++;
        }
        if (n % 2 == 1) {
            res[n/2][n/2] = n*n;
        }
        return res;
    }
}
