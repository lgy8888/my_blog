                                                  # 哈希表

                                                  # 基本原理

                                                  哈希表是根据关键码的值而直接进行访问的数据结构

                                                  关键码就是索引下标，通过下标直接访问数组中的元素

                                                  # 哈希函数

                                                  哈希表通过哈希函数将key转换成hashCode(哈希编码)

                                                  $hashCode=hashFunction(key)$

                                                  由于哈希表是有限的，所以需要通过取模运算，将hashCode映射到哈希表中

                                                  $index=hashCode%tableSize$

                                                  通过这个index将value存进哈希表中

                                                  一般哈希表用于快速判断一个元素是否出现在集合里

                                                  使用数组枚举的话时间复杂度是$O(N)$，而使用哈希表的话只需要$O(1)$

                                                  • 待查询的key通过哈希函数转换成index
                                                  • 然后直接去索引对应的位置查值

                                                  # 哈希冲突

                                                  若不同的key映射到相同的索引上，就会引发哈希冲突

                                                  解决哈希冲突可以有下面两种方法

                                                  • 开放定址法

                                                  当一个关键字和另一个关键字发生冲突时，使用某种探测技术在哈希表中形成一个探测序列，然后沿着这个探测序列依次查找下去。当碰到一个空的单元时，则插入其中。 常用的探测方法是线性探测法。 比如有一组关键字 {12，13，25，23}，采用的哈希函数为 key mod 11。当插入 12，13，25 时可以直接插入，地址分别为 1、2、3。而当插入 23 时，哈希地址为 23 mod 11 = 1。然而，地址 1 已经被占用，因此沿着地址 1 依次往下探测，直到探测到地址 4，发现为空，则将 23 插入其中。

                                                  • 拉链法

                                                  将哈希地址相同的记录存储在一张线性链表中。

                                                  # 字典法

                                                  # 基本思路

                                                  声明一个长度为26的int数组，用于记录每个字母出现了几次

                                                  遍历两个字符串，将串中的字符映射到0-25之间(str.charAt(i) - 'a')获得该字符在字典中的索引

                                                  一个循环负责++，一个循环负责--

                                                  最后遍历字典，判断字典中是否为全0

                                                  • 若全0，则代表两个字符串元素一样
                                                  • 若非全0，则代表有元素多余了

                                                  # 代码实现

                                                  class Solution {
    public boolean isAnagram(String s, String t) {
        int[] record = new int[26];
        for (int i = 0; i < s.length(); i++) {
            record[s.charAt(i) - 'a']++;
        }
        for (int i = 0; i < t.length(); i++) {
            record[t.charAt(i) - 'a']--;
        }
        for (int count: record) {
            if (count != 0) {               
                return false;
            }
        }
        return true; 
    }
}

                                                  # 基本思路

                                                  定义一个哈希表

                                                  将其中一个数组转化进去

                                                  利用set.contains()方法对第二个数组进行遍历比较

                                                  然后将结果赋到另一个哈希表里面，这一步是为了去重

                                                  使用stream对set进行变换

                                                  或者直接定义一个新的数组进行拷贝

                                                  # 代码实现

                                                  class Solution {
    public int[] intersection(int[] nums1, int[] nums2) {
        Set<Integer> set1 = new HashSet<>();
        for(int n1 : nums1){
            set1.add(n1);
        }        
        Set<Integer> res = new HashSet<>(); 
        for(int n2 : nums2){
            if(set1.contains(n2)){
                res.add(n2);
            }
        }
        
        return res.stream().mapToInt(x -> x).toArray();

                                                  # 快乐数

                                                  力扣题目链接open in new window

                                                  # 基本思路

                                                  定义一个哈希表res来接历史平方和

                                                  将给定的整数n转换为数组，用变量arr接

                                                  定义sum变量用来接平方和

                                                  遍历数组求出平方和

                                                  当平方和为1时，证明该数为快乐数

                                                  当平方和与之前算出来的平方和重复时，说明进入死循环，证明该数非快乐树

                                                  若都不符合，说明还需要继续迭代

                                                  将平方和存入历史平方和集合

                                                  用平方和更新数组

                                                  # 易错点

                                                  • 外层需要套个死循环
                                                  • 整数转数组可以先转换为String，然后利用String.charAt()来遍历字符串，再强转并存入int[]数组

                                                  # 代码实现

                                                  class Solution {
    public boolean isHappy(int n) {
        Set<Integer> res = new HashSet<>();
        int[] arr = getIntArr(n);
        int sum = 0;
        while(true){
            sum = 0;
            for(int a : arr){
                sum += a*a;
            }
            if(sum == 1) return true;
            if(res.contains(sum)) return false;
            res.add(sum);
            arr = getIntArr(sum);
        }
    }   
    
    private int[] getIntArr(Integer n){
        String str = n.toString();
		int[] arr = new int[str.length()];
		for (int i = 0; i < str.length(); i++) {
            Character ch = str.charAt(i);
            arr[i] = Integer.parseInt(ch.toString());
		}
        return arr;
    }
}

                                                  # 暴力法

                                                  # 基本思路

                                                  两层for循环，穷举两数之和，符合条件就直接返回，不符合返回null

                                                  # 代码实现

                                                  class Solution {
    public int[] twoSum(int[] nums, int target) {
        for(int i = 0; i < nums.length; i++){
            for(int j = i+1; j < nums.length; j++){
                if(nums[i] + nums[j] == target){
                    int[] res = {i, j};
                    return res;
                }
            }
        }
        return null;
    }
}

                                                  # 哈希法

                                                  # 基本思路

                                                  定义一个HashMap来存遍历结果，key存值，value存索引

                                                  每遍历一个元素就从Map里面取符合条件的值

                                                  若取到了，直接返回

                                                  没取到，将该元素放进Map中

                                                  然后继续迭代

                                                  # 代码实现

                                                  class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> res = new HashMap<>();
        for(int i = 0; i < nums.length; i++){
            Object j = res.get(target - nums[i]);
            if(j != null) {
                return new int[]{i, (int)j};
            }else{
                res.put(nums[i], i);
            }
        }
        return null;
    }
}

                                                  # 四数相加

                                                  力扣题目链接open in new window

                                                  # 基本思路

                                                  定义一个Map集合，定义一个res整数来接结果(四数相加等于0的次数)

                                                  遍历AB数组，计算AB数组两两元素之和，存入map集合中

                                                  • key存和，value存该和出现的次数

                                                  遍历CD数组，计算AB数组两两元素之和的负值，然后逐一在Map中找相等的key

                                                  • 找到则说明A+B = -(C+D)，也就是A+B+C+D=0，符合条件，加入res中

                                                  返回res

                                                  # 代码实现

                                                  class Solution {
    public int fourSumCount(int[] nums1, int[] nums2, int[] nums3, int[] nums4) {
        Map<Integer, Integer> map = new HashMap<>();
        int res = 0;
        // 计算AB数组两两元素之和，存入Map
        for(int a : nums1){
            for(int b : nums2){
                int ab = a + b;
                if(map.containsKey(ab)){
                    map.put(ab, map.get(ab)+1);
                }else{
                    map.put(ab, 1);
                }
            }
        }
        // 计算CD数组两两元素之差，与AB数组作比较
        for(int c : nums3){
            for(int d : nums4){
                int cd = -(c + d);
                if(map.containsKey(cd)){
                    res += map.get(cd);
                }
            }
        }
        return res;
    }
}

                                                  # 赎金信

                                                  力扣题目链接open in new window

                                                  # 基本思路

                                                  思路跟有效的字母异位词类似

                                                  定义一个int数组，用于映射a-z字符

                                                  遍历长串，记录字符出现的次数

                                                  遍历短串，出现一次字符就减去那个字符的计数一次

                                                  若出现负数，则说明不符合条件，返回false

                                                  没有情况则说明符合条件，返回true

                                                  # 代码实现

                                                  class Solution {
    public boolean canConstruct(String ransomNote, String magazine) {
        int[] allChar = new int[26];
        for(char c : magazine.toCharArray()){
            allChar[c-'a']++;
        }
        for(char c : ransomNote.toCharArray()){
            allChar[c-'a']--;
            if(allChar[c-'a'] < 0) return false;
        }
        return true;
    }
}

                                                  # 三数之和

                                                  力扣题目链接open in new window

                                                  # 基本思路

                                                  给定的数组先排序

                                                  确定一根指针，移动另外两根指针

                                                  利用确定的那根指针循环遍历数组

                                                  定义左指针j，右指针k

                                                  若ijk三根指针指向的数值之和<0，则移动右指针k

                                                  数值之和<0，则移动左指针j

                                                  数值之和=0，则将这三个值存入结果集合，然后同时更新左右指针，更新的同时去重

                                                  # 易错点

                                                  涉及的剪枝操作：判断第一个数是否大于0，去重

                                                  每次遍历时第一个数如果大于0，则说明后面的子数组中找不到相加等于0的元组，直接返回

                                                  题目要求不能出现相同的三元组，所以需要去重

                                                  • 第一根指针i的去重：需要判断的是以及遍历过的旧指针，不能判断没遍历的新指针，如果重复则说明当次循环没有必要进行，直接跳过
                                                  • 找到三元组时迭代左右指针后进行去重：迭代指针是必须的，去重不一定会执行，所以使用do-while循环，先迭代，然后判断迭代后是否跟没迭代时指向的数值相同，相同则继续迭代，直到迭代出的指针能指向新的值为止

                                                  # 代码实现

                                                  class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        List<List<Integer>> res = new ArrayList<>();
        Arrays.sort(nums);
        if (nums[0] > 0) return res;
        for (int i = 0; i < nums.length; i++) {
            if(nums[i] > 0) return res;
            if (i > 0 && nums[i] == nums[i - 1]) continue;
            int j = i + 1;
            int k = nums.length - 1;
            while (j < k) {
                int sum = nums[i] + nums[j] + nums[k];
                if (sum > 0) k--;
                else if (sum < 0) j++;
                else {
                    List<Integer> a = new ArrayList<>();
                    a.add(nums[i]);
                    a.add(nums[j]);
                    a.add(nums[k]);
                    res.add(a);
                    do{
                        j++;
                    } while (j < k && nums[j] == nums[j - 1]);
                    do{
                        k--;
                    } while (j < k && nums[k] == nums[k + 1]);
                }
            }
        }
        return res;
    }
}

                                                  # 四数之和

                                                  力扣题目链接open in new window

                                                  # 双指针法

                                                  # 基本思路

                                                  三数之和的外面再套一层循环

                                                  指针i控制第一层循环，指针j控制第二层循环，左指针l和右指针r共同控制第三次循环

                                                  三层循环控制四个数值之和sum，若该sum > target，则右指针左移，sum < target，则左指针右移

                                                  若sum = targe，则将其加入结果集res，然后同时移动左右指针并去重

                                                  # 易错点

                                                  涉及的剪枝操作：初始target判断，去重

                                                  每次遍历的时候判断第一个数在大于零的情况下是否还大于target，是则证明后面没有符合条件的元组了，直接返回

                                                  题目要求不能出现重复元素的四元组，所以要进行去重

                                                  • 四根指针的去重流程都是
                                                    1. 先迭代
                                                    2. 然后判断迭代后的指针指向值与迭代前指向的值是否相等
                                                    3. 相等则继续迭代，不相等则继续进入循环
                                                  • 注意：最外层指针i和中间层指针j不对初始值进行去重，特别是中间层指针j
                                                    • 至少进行了一次迭代之后才会进行去重操作

                                                  # 代码实现

                                                  class Solution {
    public List<List<Integer>> fourSum(int[] nums, int target) {
        // 定义结果集res
        List<List<Integer>> res = new ArrayList<>();
        // 给定数组排序
        Arrays.sort(nums);
        // 定义最外层循环，由指针i控制
        for (int i = 0; i < nums.length; i++) {
            // nums[i] > target 直接返回，剪枝操作
            if (nums[i] > 0 && nums[i] > target) return res;
            // 排除初始值0，对i进行去重
            if (i > 0 && nums[i] == nums[i - 1]) continue;
            // 定义中间层循环，由j控制
            for (int j = i + 1; j < nums.length; j++) {
                // 排除初始值i+1，对j进行去重
                if (j > i + 1 && nums[j] == nums[j - 1]) continue;
                // 定义左指针l和右指针r
                int l = j + 1;
                int r = nums.length - 1;
                // 定义最里层循环，由l和r控制
                while (l < r) {
                    // 计算四数之和
                    int sum = nums[i] + nums[j] + nums[l] + nums[r];
                    // 根据target和sum的大小关系决定指针的迭代
                    if (sum < target) l++;
                    else if (sum > target) r--;
                    // 找到目标四元组，加入结果集
                    else {
                        List<Integer> a = new ArrayList<>();
                        a.add(nums[i]);
                        a.add(nums[j]);
                        a.add(nums[l]);
                        a.add(nums[r]);
                        res.add(a);
                        // 对两根指针同时进行迭代并去重
                        do {
                            l++;
                        } while (l < r && nums[l] == nums[l - 1]);
                        do {
                            r--;
                        } while (l < r && nums[r] == nums[r + 1]);
                    }
                }
            }
        }
        return res;
    }
}
